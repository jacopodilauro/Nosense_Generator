@startuml

title \n\n<font color = black><size:22> SYNTAXANALYZER \n <font color = black><size:12> INTERNAL SEQUENCE DIAGRAM \n\n

participant "**SyntaxAnalyzer**" as Analyzer #dodgerblue
activate Analyzer #dodgerblue

Analyzer -[#blue]> Analyzer: crea le liste nouns, verbs, adjectives
Analyzer -[#blue]> Analyzer: crea LanguageServiceClient (try-with-resources)
Analyzer -[#blue]> Analyzer: costruisce il Document (setContent, setType, build)
Analyzer -[#blue]> Analyzer: chiama analyzeSyntax(doc)
Analyzer -[#blue]> Analyzer: ottiene la lista di Tokens

loop per ogni Token
Analyzer -[#blue]> Analyzer: getText().getContent()  // parola
Analyzer -[#blue]> Analyzer: getPartOfSpeech().getTag()  // tag grammaticale
alt tag == NOUN
Analyzer -[#blue]> Analyzer: aggiunge la parola a nouns
else tag == VERB e parola != "will"
Analyzer -[#blue]> Analyzer: aggiunge la parola a verbs
else tag == ADJ
Analyzer -[#blue]> Analyzer: aggiunge la parola a adjectives
else
Analyzer -[#blue]> Analyzer: ignora la parola
end
end

Analyzer -[#blue]> Analyzer: crea e restituisce SyntaxResult(nouns, verbs, adjectives)
deactivate Analyzer
@enduml
